58 {
59  if (thread_num > MAX_THREAD_NUM)
60  thread_num = MAX_THREAD_NUM;
61 
62  idl_thread_num = thread_num < 1 ? 1 : thread_num;
63  for (int i = 0; i < idl_thread_num; i++)
64  {
65  pool.emplace_back(
66  [this]()
67  {
68  //工作线程

86  }

120 {
121  if(stoped.load())
122  throw std::runtime_error("commit on thread pool is stoped");
123  using RetType = decltype(f(args...)); //获取函数f的返回值类型
124  auto task = std::make_shared<std::packaged_task<RetType()>>(
125  std::bind(std::forward<F>(f), std::forward<Args>(args)...)
126  );
127  //得到task 返回值

129  //添加任务

137  );
138  }
139
