59 void CoreServer::init_thread(LibeventThread *me)
60 {
61  int res;
62 
63  //建立libevent事件处理机制
64  me->p_tcp_conn = this;
65  me->base = event_base_new();
66  assert(me->base != nullptr);
67 
68  //在主线程和子线程之间建立管道
69  int fds[2];
70  res = socketpair(AF_UNIX, SOCK_STREAM, 0, fds);
71  if (res == -1)
72  {
73  perror("socketpair");
74  exit(1);
75  }
76 
77  me->notify_receive_fd = fds[0];
78  me->notify_send_fd = fds[1];
79 
80  //让子线程event base监听管道
81  event_set(&me->notify_event, me->notify_receive_fd,
82  EV_READ | EV_PERSIST, thread_process, me);
83  event_base_set(me->base, &me->notify_event);
84  res = event_add(&me->notify_event, nullptr);
85  assert( res == 0 );
86 }

107 void CoreServer::start_run()
108 {
109  evconnlistener *listener;
110 
111  //如果端口号不是EXIT_CODE，就监听该端口号
112  if(EXIT_CODE != m_port)
113  {
114  struct sockaddr_in sin;
115  memset(&sin, 0, sizeof(sin));
116  sin.sin_family = AF_INET;
117  sin.sin_port = htons(static_cast<uint16_t>(m_port));
118  std::cout << "start listen..." << std::endl;
119 
120  //监听并绑定,掉用户调函数,on_listen
121  listener = evconnlistener_new_bind(m_main_base->base,
122  on_listen, (void *) this,
123  LEV_OPT_REUSEABLE | LEV_OPT_CLOSE_ON_FREE, -1,
124  (sockaddr *) &sin, sizeof(sockaddr_in));
125  if(nullptr == listener )
126  {
127  std::cerr << "listen error " << strerror(errno) << std::endl;
128  exit(1);
129  }
130  }
131 
132  //开启各个子线程
133  for(int i=0; i < m_thread_num; i++)
134  {
135  pool->enqueue(worker_thread, &m_threads[i]);
136 
137  }
138 
139  //开启主线程的事件循环
140  event_base_dispatch(m_main_base->base);
141 
142  //事件循环结果，释放监听者的内存
143  if( m_port != EXIT_CODE )
144  {
145  std::cout << "release listen" << std::endl;
146  evconnlistener_free(listener);
147  }
148 
149 
150 }

195 void CoreServer::thread_process(int fd, short which, void *arg)
196 {
197  auto *me = static_cast<LibeventThread*>(arg);
198 
199  //从管道中读取数据（socket的描述符或操作码）
200  int pipefd = me->notify_receive_fd;
201  evutil_socket_t confd;
202  read(pipefd, &confd, sizeof(evutil_socket_t));
203  //如果操作码是EXIT_CODE，则终于事件循环
204  if( EXIT_CODE == confd )
205  {
206  event_base_loopbreak(me->base);
207  return;
208  }
209 
210  //新建连接
211  struct bufferevent *bev;
212  bev = bufferevent_socket_new(me->base, confd, BEV_OPT_CLOSE_ON_FREE);
213  if (!bev)
214  {
215  std::cerr << "Error constructing bufferevent!" << std::endl;
216  event_base_loopbreak(me->base);
217  return;
218  }
219 
220  std::cout << "confd " << confd << std::endl;
221  //将该链接放入队列
222  auto *tmp = new Conn(confd);
223  tmp->m_thread = me;
224  me->conn.push_front(tmp);
225  //准备从socket中读写数据
226  std::cout << "conn size " << me->conn.size() << std::endl;
227  std::cout << "conn first " << me->conn.front()->m_fd << std::endl;
228  bufferevent_setcb(bev, on_read, on_write, on_close, me->conn.front());
229  bufferevent_enable(bev, EV_WRITE);
230  bufferevent_enable(bev, EV_READ);
231 
232  //调用用户自定义的连接事件处理函数
233  me->p_tcp_conn->connect_cb(me->conn.front());
234 }
